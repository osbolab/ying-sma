203  WriteLock claim()
204  {
205  // Try until we find a slot not being written to.
206  // This may occur if every slot is being written to, in which case we busy
207  // loop waiting for one to open.
208  mono_type seq;
209  Entry* entry;
210  do {
211  // Claim the next sequence number and the according buffer slot.
212  seq = next_write++;
213  // The sequence numbers outrun the buffer indices by one because the
214  // readers need to start behind the first writer.
215  entry = entries + ((seq - 1) & idx_mask);
216  // If another writer set the flag then find a new slot.
217  } while (entry->writing.test_and_set());
218  // We have set the writing flag
219 
220  if (entry->readers != 0)
221  throw std::runtime_error(
222  "Ring buffer overrun (the next slot to write is being read).");
223 
224  return WriteLock(*entry);
225  }

227  std::pair<ReadLock, bool> try_pop()
228  {
229  // Pretend to claim the next read position and, if we're successful,
230  // CAS until we actually commit it.
231  mono_type seq = next_read;
232  Entry* entry;
233  while (true) {
234  entry = entries + (seq & idx_mask);
235  ++entry->readers;
236  // Writers are now blocked from claiming it, but one may have it already.
237  if (entry->writing.test_and_set())
238  return std::make_pair(ReadLock(), false);
239 
240  entry->writing.clear();
241 
242  // Since we just pretended to increment the read position (in case we
243  // failed), someone else might have succeeded while we tried.
244  if (next_read.compare_exchange_weak(seq, seq + 1))
245  break;
246  // We failed the CAS; remove ourselves from the entry we chose and retry.
247  --entry->readers;
248  }
249

251  }
